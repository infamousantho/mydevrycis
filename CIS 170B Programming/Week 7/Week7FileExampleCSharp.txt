        private void btnPrintDirectory_Click(object sender, EventArgs e)
        {
            string dir = Directory.GetCurrentDirectory();
            string msg = "";
            msg += dir;
            string[] files = Directory.GetFiles(dir);
            foreach (string val in files)
            {
                msg += val;
                msg += "\n";

            }
            MessageBox.Show(msg);

        }
